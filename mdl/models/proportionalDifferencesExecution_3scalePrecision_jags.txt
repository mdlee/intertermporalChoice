# PROPORTIONAL DIFFERENCES
# execution
model{
  for (i in 1:nParticipants){
    # participant-specific parameters
    epsilon[i] ~ dunif(0, 0.5)
    delta[i] ~ dnorm(0, precision)
    for (j in 1:nTrials){
      # differences
      diffRewardR[i,j] = (rLL[i, j] - rSS[i, j])/rLL[i, j]
      diffTimeR[i,j] = (tLL[i, j] - tSS[i, j])/tLL[i, j]
      # probability of choosing larger later
      theta[i, j] = ifelse(step(diffRewardR[i, j] - diffTimeR[i, j] - delta[i]), 1-epsilon[i], epsilon[i])
      # decision
      decision[i, j] ~ dbern(theta[i, j])   
    }
  }    
}
