# DIRECT DIFFERENCES
# v3 lognormal for truncated normal
# supply precision
model{
  for (i in 1:nParticipants){
    # participant-specific parameters
    sigma[i] ~ dlnorm(0, precision)
    delta[i] ~ dnorm(0, precision)
    omega[i] ~ dunif(0, 1)
    for (j in 1:nTrials){
      # differences
      diffRewardR[i,j] = omega[i]*(rLL[i, j] - rSS[i, j])
      diffTimeR[i,j] = (1 - omega[i])*(tLL[i, j] - tSS[i, j])
      # probability of choosing larger later
      theta[i, j] = phi((diffRewardR[i, j] - diffTimeR[i, j] - delta[i])/sigma[i])
      # decision
      decision[i, j] ~ dbern(theta[i, j])   
    }
  }    
}
